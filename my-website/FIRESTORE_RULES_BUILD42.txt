rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() { return request.auth != null; }
    function isAdmin() { return signedIn() && request.auth.token.email == 'emviemsalva@gmail.com'; }

    match /contentRequests/{id} {
      allow create: if signedIn() && request.resource.data.uid == request.auth.uid;
      allow read: if isAdmin() || (signedIn() && resource.data.uid == request.auth.uid);
      allow update, delete: if isAdmin();
    }
    match /brokenLinkReports/{id} {
      allow create: if signedIn() && request.resource.data.uid == request.auth.uid;
      allow read: if isAdmin() || (signedIn() && resource.data.uid == request.auth.uid);
      allow update, delete: if isAdmin();
    }
    match /cineflexNotifications/{id} {
      allow read: if signedIn();
      allow create, update, delete: if isAdmin();
    }
    match /importedContent/{id} {
      allow read: if signedIn();
      allow create, update, delete: if isAdmin();
    }

    // Keep and merge your existing rules for profiles, comments, ratings,
    // Watch Party rooms, progression, and CineFlex homepage configuration.
  }
}
