// Merge these rules into your existing Firestore rules.
// They require authenticated users for Watch Party rooms and chat.
match /watchPartyRooms/{roomId} {
  allow read: if request.auth != null;
  allow create: if request.auth != null && request.resource.data.hostUid == request.auth.uid;
  allow update, delete: if request.auth != null && resource.data.hostUid == request.auth.uid;

  match /members/{uid} {
    allow read: if request.auth != null;
    allow create, update, delete: if request.auth != null && request.auth.uid == uid;
  }

  match /messages/{messageId} {
    allow read: if request.auth != null;
    allow create: if request.auth != null
      && request.resource.data.uid == request.auth.uid
      && request.resource.data.text is string
      && request.resource.data.text.size() <= 280;
    allow update, delete: if false;
  }
}
