match /users/{uid}/membership/{documentId} {
  // A signed-in user may read only their own membership status.
  allow read: if request.auth != null && request.auth.uid == uid;

  // Only the owner/admin should grant, extend, or remove VIP.
  allow create, update, delete: if isOwner();
}

// }
