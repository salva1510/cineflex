match /users/{uid}/watchProgress/{progressId} {
  allow read, create, update, delete: if request.auth != null && request.auth.uid == uid;
}

match /livePresence/{sessionId} {
  allow read: if true;
  allow create, update: if request.resource.data.lastSeenMs is int
    && request.resource.data.mediaKey is string;
  allow delete: if request.auth != null;
}
